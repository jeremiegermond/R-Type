39 void insert(entity_type id, const Component &value) {
40 size_t sparseIndex = inBlockIndex(id);
41
42 if (id == size && id == blocks.size() * BLOCK_SIZE) {

62 void erase(entity_type id) {
63 size_t index = inBlockIndex(id);
64 Block *block = this->blocks[blockIndex(id)];
65 entity_type previousSparseValue = block->sparse[index];
66
67 if (id == this->max) {
68 updateMax();
69 if (blocks.size() > 1 && max < BLOCK_SIZE / 2 + (blocks.size() - 2) * BLOCK_SIZE) {
70 delete this->blocks[this->blocks.size() - 1];
71 this->blocks.pop_back();
72 }
73 }
74
75 if (previousSparseValue != NULL_ENTITY) {
76 block->sparse[index] = NULL_ENTITY;
77 }
78
79 block->sparse[index] = block->dense[index];
80 entity_type *lastDensePtr = &this->blocks[blockIndex(this->size - 1)]->dense[inBlockIndex(this->size - 1)];
81 if (*lastDensePtr == id)
82 return;
83 block->dense[index] = *lastDensePtr;
84 *lastDensePtr = id;
85 block->sparse[inBlockIndex(*lastDensePtr)] = index;
86 --size;
87 }
